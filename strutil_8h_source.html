347 {
348 std::vector<std::string> tokens;

350
351 std::string token;

354 tokens.push_back(token);

372 static inline std::vector<std::string> split(const std::string & str, const std::string & delim)
373 {
374 size_t pos_start = 0, pos_end, delim_len = delim.length();
375 std::string token;
376 std::vector<std::string> tokens;
377
378 while ((pos_end = str.find(delim, pos_start)) != std::string::npos)
379 {
380 token = str.substr(pos_start, pos_end - pos_start);
381 pos_start = pos_end + delim_len;
382 tokens.push_back(token);
383 }
384
385 tokens.push_back(str.substr(pos_start));
386 return tokens;

417 {

419 std::string tstr = src + " ";

421 std::sregex_token_iterator niter(tstr.begin(), tstr.end(), rgx);
